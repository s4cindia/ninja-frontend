**1. First, create the logo as an SVG or download it:**

Create `public/s4carlisle-logo.png` by downloading from the company website, or create a text-based logo component.

**2. Create `src/components/ui/Logo.tsx`:**
```typescript
interface LogoProps {
  size?: 'sm' | 'md' | 'lg';
  showText?: boolean;
  className?: string;
}

export function Logo({ size = 'md', showText = true, className = '' }: LogoProps) {
  const sizes = {
    sm: 'h-6',
    md: 'h-8',
    lg: 'h-12',
  };

  return (
    <div className={`flex items-center gap-2 ${className}`}>
      <img 
        src="/s4carlisle-logo.png" 
        alt="S4Carlisle Publishing Services" 
        className={`${sizes[size]} w-auto object-contain`}
      />
      {showText && (
        <div className="hidden sm:block">
          <span className="text-sm text-gray-500">powered by</span>
        </div>
      )}
    </div>
  );
}
```
